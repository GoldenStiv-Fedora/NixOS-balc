     # 1. Устанавливаем инструменты
     sudo nix-env -iA nixos.git nixos.rsync && \
    
     # 2. Создаем структуру и инициализируем репозиторий БЕЗ скачивания файлов
     sudo mkdir -p /etc/nixos/.sync && cd /etc/nixos/.sync && \
     sudo git init && \
     sudo git remote add origin https://github.com/GoldenStiv-Fedora/NixOS-balc.git && \
    
     # 3. Включаем выборочное скачивание (sparse-checkout) только для папки configs
    sudo git config core.sparseCheckout true && \
    echo "configs/*" | sudo tee .git/info/sparse-checkout && \
   
    # 4. Скачиваем только нужную папку (глубина 1 — без истории коммитов)
    sudo git pull --depth 1 origin main && \
   
    # 5. Копируем конфиги (с защитой hardware-config)
    sudo rsync -a configs/ /etc/nixos/ --exclude=hardware-configuration.nix && \
   
    # 6. Собираем систему
    sudo nixos-rebuild switch --flake /etc/nixos#nixos --option experimental-features "nix-command flakes" && \
   
    # 7. Финальный запуск синхронизации (она уже докачает картинки и всё остальное)
    sudo nixos-git-sync
